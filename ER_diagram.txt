users[icon: user, color: Green]{
  id serial pk
  name string
  email string UNIQUE
  phone string
  password string
  role enum[trainer,client] Default client

  createdAt timestamp
  updatedAt timestamp
}

fitness_subscriptions[icon: clipboard, color: Red]{
  id serial pk
  trainerId int fk
  diet_plan_id int fk
  workout_plan_id int fk
  price int 
  description text 
}

  lunch string
  sanck string
  dinner string

  createdAt timestamp
  updatedAt timestamp
}

workout_plans[icon: dumbbell, color: Orange]{

  saturday_plan string

  updatedAt timestamp
}

fitness_subscriptions.trainerId > users.id
fitness_subscriptions.diet_plan_id - diet_plans.id
fitness_subscriptions.workout_plan_id - workout_plans.id

diet_plans.createdBy > users.id
workout_plans.createdBy > users.id

session_schedules.subscription_planId > fitness_subscriptions.id
